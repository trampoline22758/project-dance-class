/* ====== 2) Utilities ============================================================ */ const DAYS = ["Mon","Tue","Wed","Thu","Fri","Sat","Sun"]; const timesSorted = (a,b)=>a.localeCompare(b); const fmtTime = t => { // "16:00" -> "4:00 PM" const [h,m]=t.split(":").map(Number); const d=new Date(); d.setHours(h,m); return d.toLocaleTimeString([], {hour:"numeric", minute:"2-digit"}); }; const unique = (arr,key) => [...new Set(arr.map(x=>x[key]))].filter(Boolean); /* ====== 3) Controls ============================================================= */ const state = {day:null, q:"", studio:"", teacher:"", age:""}; const dayChips = document.getElementById("dayChips"); const studioSelect = document.getElementById("studioSelect"); const teacherSelect = document.getElementById("teacherSelect"); const ageSelect = document.getElementById("ageSelect"); const q = document.getElementById("q"); const resetBtn = document.getElementById("resetBtn"); const printBtn = document.getElementById("printBtn"); DAYS.forEach(d=>{ const b=document.createElement("button"); b.className="chip"; b.setAttribute("role","tab"); b.setAttribute("aria-pressed","false"); b.textContent=d; b.addEventListener("click",()=>{ state.day = (state.day===d)? null : d; [...dayChips.children].forEach(el=>el.setAttribute("aria-pressed", String(el.textContent===state.day))); render(); }); dayChips.appendChild(b); }); unique(CLASSES,"studio").sort().forEach(s=>{ const o=document.createElement("option"); o.value=s; o.textContent=s; studioSelect.appendChild(o); }); unique(CLASSES,"teacher").sort().forEach(t=>{ const o=document.createElement("option"); o.value=t; o.textContent=t; teacherSelect.appendChild(o); }); q.addEventListener("input", ()=>{state.q=q.value.trim().toLowerCase(); render();}); studioSelect.addEventListener("change", ()=>{state.studio=studioSelect.value; render();}); teacherSelect.addEventListener("change", ()=>{state.teacher=teacherSelect.value; render();}); ageSelect.addEventListener("change", ()=>{state.age=ageSelect.value; render();}); resetBtn.addEventListener("click", ()=>{ state.day=null; state.q=""; state.studio=""; state.teacher=""; state.age=""; q.value=""; studioSelect.value=""; teacherSelect.value=""; ageSelect.value=""; [...dayChips.children].forEach(el=>el.setAttribute("aria-pressed","false")); render(); }); printBtn.addEventListener("click", ()=>window.print()); /* ====== 4) Rendering ============================================================ */ const gridBody = document.getElementById("gridBody"); const mobileList = document.getElementById("mobileList"); const emptyState = document.getElementById("emptyState"); function filterData(list){ return list.filter(c=>{ if(state.day && c.day!==state.day) return false; if(state.studio && c.studio!==state.studio) return false; if(state.teacher && c.teacher!==state.teacher) return false; if(state.age && c.age!==state.age) return false; if(state.q){ const s=(c.title+" "+c.style+" "+c.teacher+" "+c.studio+" "+c.age).toLowerCase(); if(!s.includes(state.q)) return false; } return true; }); } function renderGrid(data){ // Build rows by unique time slots const times=[...new Set(data.map(c=>c.start))].sort(timesSorted); const byDay = d => data.filter(x=>x.day===d).sort((a,b)=>a.start.localeCompare(b.start)); gridBody.innerHTML=""; if(!data.length){ gridBody.innerHTML=""; return; } times.forEach(t=>{ const tr=document.createElement("tr"); const timeTd=document.createElement("td"); timeTd.textContent=`${fmtTime(t)}`; tr.appendChild(timeTd); DAYS.slice(0,6).forEach(d=>{ const td=document.createElement("td"); const classes = byDay(d).filter(x=>x.start===t); classes.forEach(c=>{ const div=document.createElement("div"); div.className="slot"; div.innerHTML = `
${c.title}
${fmtTime(c.start)}–${fmtTime(c.end)} • ${c.studio}
Instructor: ${c.teacher} • Age: ${c.age}
${c.style} `; td.appendChild(div); }); if(!td.children.length){ td.innerHTML = "—"; } tr.appendChild(td); }); gridBody.appendChild(tr); }); } function renderMobile(data){ mobileList.innerHTML=""; const ordered = data.slice().sort((a,b)=>{ const di = DAYS.indexOf(a.day)-DAYS.indexOf(b.day); return di || a.start.localeCompare(b.start); }); ordered.forEach(c=>{ const card=document.createElement("article"); card.className="card"; card.innerHTML=`

${c.title}

${c.day} • ${fmtTime(c.start)}–${fmtTime(c.end)} • ${c.studio}
Instructor: ${c.teacher} • Age: ${c.age}
${c.style}
`; mobileList.appendChild(card); }); } function render(){ const filtered = filterData(CLASSES); emptyState.hidden = filtered.length>0; renderGrid(filtered); renderMobile(filtered); } render();
